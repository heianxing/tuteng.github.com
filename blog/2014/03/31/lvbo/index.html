均值滤波

Mar 31st, 2014 

%滤波函数
function [s] = aver2(a)
%s=0;
x=2;
y=2;
s = (a(x-1,y-1)+a(x-1,y)+a(x-1,y+1)+a(x,y-1)+a(x,y)+a(x,y+1)+a(x+1,y-1)+a(x+1,y)+a(x+1,y+1));
%滤波公式，本次使用的3x3的模板
s=s/9;


%主函数如下
%均值滤波
pic=imread('H:\matlab\zuoye.jpg');
figure;
imshow(pic);
i=rgb2gray(pic);  %将图像转换为灰度图，便于处理
figure;
imshow(i);
img3=imnoise(i,'gaussian');  %添加高斯噪声
figure;
imshow(img3)
img2=imnoise(i,'salt & pepper');  %添加椒盐噪声
figure;
imshow(img2)
img1=img3;
img4=img2;

[len3,len4]=size(img4);  %获得椒盐噪声图像矩阵的行数和列列数

len3=len3-2;
len4=len4-2;

[len1,len2]=size(img1);  %获得添加了高斯噪声图像的行数和列数
%leng2=img2-2;
len1=len1-2;
len2=len2-2;
m=zeros(3,3);  %初始化一个矩阵，以备模板使用


for i=1:len1
    for j=1:len2
        %获得模板
        m(1,1)=img1(i,j);
        m(1,2)=img1(i,j+1);
        m(1,3)=img1(i,j+2);
        m(2,1)=img1(i+1,j);
        m(2,2)=img1(i+1,j+1);
        m(2,3)=img1(i+1,j+2);
        m(3,1)=img1(i+2,j);
        m(3,2)=img1(i+2,j+1);
        m(3,3)=img1(i+2,j+2);
        s2=aver2(m);
        img3(i+1,j+1)=s2;
    end
end
figure;
imshow(img3)

for i=1:len3
    for j=1:len4
        %获得模板
        m(1,1)=img4(i,j);
        m(1,2)=img4(i,j+1);
        m(1,3)=img4(i,j+2);
        m(2,1)=img4(i+1,j);
        m(2,2)=img4(i+1,j+1);
        m(2,3)=img4(i+1,j+2);
        m(3,1)=img4(i+2,j);
        m(3,2)=img4(i+2,j+1);
        m(3,3)=img4(i+2,j+2);   
        s3=aver2(m);  %均值滤波
        img2(i+1,j+1)=s3;
    end
end
figure;
imshow(img2);
