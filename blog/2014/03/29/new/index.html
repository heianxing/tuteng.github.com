二分查找

Mar 29th, 2014 

#include <iostream>
using namespace std;
int search(int arr[],int k,int n);
int main()
{

int arr[15]={0,1,2,2,2,2,3,3,3,4,5,6,7,8,9};
int a=search(arr,2,15);
if(a==-1)
    cout<<"出错 "<<endl;
else 
    cout<<"数组下标："<<a<<endl;
return 0;
}

int search(int arr[],int k,int n)
{
    int start,end,mid;
    start=0;
    end=n-1;
    while(start<=end)   //判断条件
    {
        mid=(start+end)/2;  //取中间值
        if(k==arr[mid])
        {
            while(k==arr[mid])
            {
                mid++;
            }
    /*
        加上while可以获取最右边的第一个出现的索引，不加while为最左边第一次出现的索引
    */
            return mid-1;
        }
        else if(k<arr[mid])  //继续往左查找
            end=mid-1;
        else
            start=mid+1;  //往右查找

    }
return -1;
}
