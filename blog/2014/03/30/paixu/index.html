插入排序

Mar 30th, 2014 | Comments 

#include <iostream>
using namespace std;
//参数为数组和其大小
void insert(int *data,int n)  
{
    int target;
    for(int i=1; i<n; i++) 
    {
        int j = i;
        target = data[i]; //将要排序的元素
        while(j > 0 && target < data[j-1])  //排序的条件，j大于0，并且将要排序的元素要找到从左边起第一个大于它的元素，放在这个元素的左边，
        {
            data[j] = data[j-1];
            j--;
            //后边的元素整体后移一个
        }
        data[j] = target;
    }
    for(int i = 0;i < n; i++)
        cout << data[i] <<",";
}

void main()
{
    int data[]={6,4,2,3,1,7,9};
    int n=7;
    cout << "排序前：";
    for(int i = 0;i < 7; i++)
        cout << data[i] <<  ",";

    cout << "排序后：";
    insert(data,n);
    cout << endl;
}
//平均时间复杂度O(n2)
