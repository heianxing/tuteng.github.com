#include <iostream>
using namespace std;
//参数为数组和其大小
void insert(int *data,int n)  
{
    int target;
    for(int i=1; i<n; i++) 
    {
        int j = i;
        target = data[i]; //将要排序的元素
        while(j > 0 && target < data[j-1])  //排序的条件，j大于0，并且将要排序的元素要找到从左边起第一个大于它的元素，放在这个元素的左边，
        {
            data[j] = data[j-1];
            j--;
            //后边的元素整体后移一个
        }
        data[j] = target;
    }
    for(int i = 0;i < n; i++)
        cout << data[i] <<",";
}

void main()
{
    int data[]={6,4,2,3,1,7,9};
    int n=7;
    cout << "排序前：";
    for(int i = 0;i < 7; i++)
        cout << data[i] <<  ",";

    cout << "排序后：";
    insert(data,n);
    cout << endl;
}
//平均时间复杂度O(n2)

#include <iostream>
using namespace std;
int search(int arr[],int k,int n);
int main()
{

int arr[15]={0,1,2,2,2,2,3,3,3,4,5,6,7,8,9};
int a=search(arr,2,15);
if(a==-1)
    cout<<"出错 "<<endl;
else 
    cout<<"数组下标："<<a<<endl;
return 0;
}

int search(int arr[],int k,int n)
{
    int start,end,mid;
    start=0;
    end=n-1;
    while(start<=end)   //判断条件
    {
        mid=(start+end)/2;  //取中间值
        if(k==arr[mid])
        {
            while(k==arr[mid])
            {
                mid++;
            }
    /*
        加上while可以获取最右边的第一个出现的索引，不加while为最左边第一次出现的索引
    */
            return mid-1;
        }
        else if(k<arr[mid])  //继续往左查找
            end=mid-1;
        else
            start=mid+1;  //往右查找

    }
return -1;
}
